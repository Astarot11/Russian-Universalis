estate_hhh_disaster = {
	potential = {
		has_dlc = "The Cossacks"
		has_estate = estate_hhh
		estate_influence = {
			estate = estate_hhh
			influence = 80
		}
		OR = {
			NOT = { has_country_flag = hhh_estate_triggered }
			had_country_flag = {
				flag = cossack_estate_hhh
				days = 365
			}
		}
	}


	can_start = {
		has_any_disaster = no
		estate_influence = {
			estate = estate_hhh
			influence = 100
		}
	}
	
	can_stop = {
		OR = {
			has_any_disaster = yes
			NOT = {
				estate_influence = {
					estate = estate_hhh
					influence = 100
				}		
			}
		}
	}
	
	down_progress = {
		factor = 1
	}
	
	progress = {
		modifier = {
			factor = 5
			estate_influence = {
				estate = estate_hhh
				influence = 100
			}
		}		
	}
	
	can_end = {
		always = yes
	}

	on_start = hhh_estate_disasters.1
	
	on_monthly = {
	}
}
