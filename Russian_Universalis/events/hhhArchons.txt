			has_ruler_modifier = Archon_dip
			has_ruler_modifier = Archon_mil
			has_country_flag = event_archon_flag
		}

	immediate = {
		hidden_effect = {

				}
		}
	}
	
	option = {
		name = chosearch.a

		        clr_country_flag = event_archon_flag
		 }
		change_adm = 1

				name = "Archon_adm"
				duration = -1
			}
	}

		name = chosearch.b
		 hidden_effect = {
		        clr_country_flag = event_archon_flag
		                }
		change_dip = 1
		add_ruler_modifier = {
